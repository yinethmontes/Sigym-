SELECT A.DEP_IDE AS IDENTIFICACION, DEP_NOM AS NOMBRES, DEP_APE AS APELLIDOS, 
B.PLE_IDE AS NUMERO_PLAN, B.PLE_FEI AS FECHA_INICIO, PLE_FEF AS FECHA_FINAL,
C.DPE_DIA AS DIA, DPE_SER AS SERIES, DPE_REP AS REPETICIONES,
D.EJE_NOM AS EJERCICIO, E.MAQ_NOM AS MAQUINA FROM DEPORTISTA A,
PLANENTRENAMIENTO B,DETALLEPE C, EJERCICIO D, MAQUINA E
WHERE A.DEP_IDE = 93378156 AND
A.DEP_IDE = B.DEP_IDE AND
B.PLE_IDE = C.PLE_IDE AND
C.EJE_IDE = D.EJE_IDE AND
D.MAQ_IDE = E.MAQ_IDE
